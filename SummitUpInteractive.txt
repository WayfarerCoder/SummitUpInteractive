        function wrapLabel(label) {
            const maxLength = 16;
            if (typeof label !== 'string' || label.length <= maxLength) {
                return label;
            }
            const words = label.split(' ');
            const lines = [];
            let currentLine = '';
            for (const word of words) {
                if ((currentLine + word).length > maxLength) {
                    lines.push(currentLine.trim());
                    currentLine = '';
                }
                currentLine += word + ' ';
            }
            lines.push(currentLine.trim());
            return lines;
        }

        const tooltipTitleCallback = function(tooltipItems) {
            const item = tooltipItems[0];
            let label = item.chart.data.labels[item.dataIndex];
            if (Array.isArray(label)) {
                return label.join(' ');
            } else {
                return label;
            }
        };

        document.addEventListener('DOMContentLoaded', () => {
            const services = [
                { name: 'Company Formation' },
                { name: 'Offshore Accounts' },
                { name: 'Business Loans' },
                { name: 'Mortgage Services' },
                { name: 'Accounting & VAT' },
                { name: 'Corporate Taxation' },
                { name: 'POS Merchant Loans' },
                { name: 'Bank Guarantees' }
            ];
            
            const container = document.querySelector('.hub-spoke-container');
            const radius = 230; 
            const angleStep = (2 * Math.PI) / services.length;

            services.forEach((service, i) => {
                const spoke = document.createElement('div');
                spoke.className = 'spoke bg-[#D4AF37] text-black font-semibold shadow-md';
                spoke.textContent = service.name;
                container.appendChild(spoke);
                spoke.style.setProperty('--scale-factor', 1.15);

                const line = document.createElement('div');
                line.className = 'line';
                container.appendChild(line);

                const updatePositions = () => {
                    const isMobile = window.innerWidth < 767;

                    if (!isMobile) {
                        const angle = i * angleStep - (Math.PI / 2); // Start from top
                        const x = radius * Math.cos(angle);
                        const y = radius * Math.sin(angle);
                        
                        spoke.style.transform = `translate(${x}px, ${y}px)`;

                        const lineAngle = Math.atan2(y, x) * (180 / Math.PI);
                        const distance = Math.sqrt(x*x + y*y) - 65; 

                        line.style.width = `${distance}px`;
                        line.style.transform = `rotate(${lineAngle}deg)`;
                        line.style.left = '50%';
                        line.style.top = '50%';
                        line.style.display = 'block';

                        spoke.onmouseover = () => { spoke.style.transform = `translate(${x}px, ${y}px) scale(1.15)`; };
                        spoke.onmouseout = () => { spoke.style.transform = `translate(${x}px, ${y}px) scale(1)`; };

                    } else {
                        spoke.style.transform = 'translate(0, 0)';
                        line.style.display = 'none';
                        spoke.onmouseover = null;
                        spoke.onmouseout = null;
                    }
                };
                
                updatePositions();
                window.addEventListener('resize', updatePositions);
            });
        });
